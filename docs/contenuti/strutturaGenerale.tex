\section{Struttura Generale}
\subsection{Pagine Pubbliche}
\begin{itemize}
    \item \textbf{Homepage}: la prima pagina visualizzata accedendo al sito.\\
    Si compone di una griglia di fotografie divise per categoria di appartenenza. La griglia ha la caratteristica di preservare l'aspect-ratio delle immagini creando un layout particolare;
    \item \textbf{About}: pagina informativa che racconta la storia del fotografo, la sua carriera e il suo approccio artistico;
    \item \textbf{Contact}: Pagina con un modulo per inviare messaggi diretti al fotografo, comprendente campi per nome, email e messaggio.
    \item \textbf{Services}: pagina dove sono presentati i servizi a pagamento del fotografo.
\end{itemize}

\subsection{Pagine Utenti Loggati}
\begin{itemize}
\item \textbf{Prenotazione Sevizio}: pagina dedicata agli utenti registrati e autenticati, dove \`e possibile selezionare data e orario per prenotare un servizio a pagamento con il fotografo.
\end{itemize}

\subsection{Pagine Amministratore}
\begin{itemize}
\item \textbf{Dashboard}: pagina dedicata alla gestione delle fotografie presenti nella Homepage. \`E possibile eliminarle, ``nasconderle" o modificarne le informazioni testuali quali titolo, descrizione, data e luogo dello scatto;
\item \textbf{Inserimento Immagini}: pagina per il caricamento delle nuove immagini in Homepage;
\item \textbf{Gestione Prenotazioni}: Pagina per visualizzare, confermare o annullare le prenotazioni ricevute dagli utenti registrati.
\end{itemize}
